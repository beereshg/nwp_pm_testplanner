def bin2float(binary, fmt):
    '''Convert binary number into float or signed int32 format using the provided format definition

    Args:
        binary (str): pass the value to be converted from a float or signed int32 format
        fmt (str): Fixed floating point format, 'float', 'double' and 'int32' (signed int32)

    Format Syntax::

        S16.2.13
        ^ ^   ^
        | |   +- exponent
        | +----- significand size
        | +----- size
        +------- Signed / Unsigned
    '''
    binary = int(binary)
    if (type(binary) != int and type(binary) != int):
        raise TypeError("Input data is not a int: %s" % (type(binary)))

    if fmt == 'float':
        return struct.unpack('!f',struct.pack('!I', binary))[0]
    elif fmt == 'double':
        return struct.unpack('!d',struct.pack('!Q', binary))[0]
    elif fmt == 'int32':
        return struct.unpack('!i',struct.pack('!I', binary))[0]
    elif ":" in fmt:
        return binary * float(fmt.split(':')[1])
    else:
        if len(fmt.split('.')) == 2:
            m = re.search(r"(S|U|u|s)(-?\d+)\.(-?\d+)", fmt)
        else:
            m = re.search(r"(S|U|u|s)(-?\d+)\.(-?\d+)\.(-?\d+)", fmt)

        signed = (m.group(1) == "S")

        if len(fmt.split('.')) == 2:
            size = (abs(int(m.group(2))) + int(m.group(3)))
            exponent = int(m.group(3))
        else:
            size = int(m.group(2))
            exponent = int(m.group(4))

        if (signed and (binary >> (size - 1)) == 1):
            # 2's complement adjustment to make this number positive
            mask = (1 << size) - 1
            binary = ((~binary) + 1) & mask
            data = -float(binary) / pow(2.0, exponent)
        else:
            data = float(binary) / pow(2.0, exponent)

        return data


class tpmi:
    def __init__(self, die):
        self.die = die
        
        if "cbb" in die.name:
            self.path = die.base.tpmi
        else:
            self.path = die.punit.ptpcfsms.ptpcfsms
            
        self.update_regs()
        
    def update_regs(self):
        self.ufs_header = self.path.ufs_header.read()
