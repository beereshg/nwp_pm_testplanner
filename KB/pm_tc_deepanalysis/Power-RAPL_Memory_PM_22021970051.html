#!/usr/bin/env python3
"""
nwp_memory_pm.py — NWP PSS: MemClos + MR4 CLTT Memory PM Tests

Covers:
  - [PSS][MEMORY] Mclos0 Throttling Status       (HSD 22021970053)
  - [PSS][MEMORY] MemClos Feature Discovery       (HSD 22021970057)
  - [PSS][MEMORY] MR4-based CLTT                  (HSD 22021970051)

NWP memory PM via TPMI and OS Mailbox:
  - NWP: 1 NIO with LPDDR6 memory controller
  - MemClos: SST-BF memory QoS via PQR_ASSOC (MSR 0xC8F) and OS mailbox
  - MR4 CLTT: Closed-Loop Thermal Throttling via memory temperature reads
  - TPMI memory PM registers at imh.punit.ptpcfsms

Execution environment: HSLE
"""

import time
import namednodes
from svtools.common import baseaccess
import pss_common as psc

base = baseaccess.getglobalbase()
sv = namednodes.sv.get_manager(['socket'])
sv.get_all(stop_on_error=True)

try:
    import cli
except ModuleNotFoundError:
    print("PM_INFO :: Could not import CLI. Unable to use any simics cli commands.")

try:
    cli.run_command("emu.engine.wait-for-cycle -relative 1000")
except NameError:
    pass
except cli.errors.CliError:
    print("PM_INFO :: Unable to use wait-for-cycle commands. Expected if you are in VP.")

ACCESS_METHOD = baseaccess.getaccess()
if hasattr(baseaccess.CFG, "emulation"):
    ACCESS_METHOD = "emulation"

NWP_NUM_CBBS = 2

# OS Mailbox commands for memory PM
OSMB_READ_PKG_CONFIG = 0x78
OSMB_WRITE_PKG_CONFIG = 0x79

# MemClos sub-commands
MEMCLOS_GET_CONFIG = 0x00
MEMCLOS_SET_CONFIG = 0x01


def wait(interface):
    wait_time = 0
    while interface.run_busy.read():
        if ACCESS_METHOD == "stub":
            interface.write(0)
        elif ACCESS_METHOD == "emulation":
            try:
                cli.run_command(f"emu.engine.wait-for-cycle -relative {100000 + wait_time * 10000}")
            except (NameError, AttributeError):
                time.sleep(1)
        else:
            time.sleep(1)
        wait_time += 1
        if wait_time > 20:
            return -1
    return 0


def os_mailbox(socket_number=0, imh=0, command=0, sub_command=0, parameter=0, data=None):
    """Send OS-to-Pcode mailbox command."""
    socket = namednodes.sv.socket.getBySocket(socket_number)
    mb_iface = socket.getbypath(f"imh{imh}").punit.ptpcfsms.ptpcfsms.os_mailbox_interface
    mb_data  = socket.getbypath(f"imh{imh}").punit.ptpcfsms.ptpcfsms.os_mailbox_data

    wait(mb_iface)

    if data is None:
        mb_data.write(0)
    else:
        mb_data.write(data)

    mb_iface.command.store(command)
    mb_iface.sub_command.store(sub_command)
    mb_iface.parameter.store(parameter)
    mb_iface.run_busy.store(1)
    mb_iface.flush()

    wait(mb_iface)

    return_code = mb_iface.command.read()
    return_data = mb_data.read()
    return return_data, return_code


def test_memclos_discovery():
    """Test MemClos feature discovery — HSD 22021970057."""
    print("=" * 65)
    print("TEST: MemClos Feature Discovery (HSD 22021970057)")
    print("=" * 65)

    # Read PQR_ASSOC MSR (0xC8F) to check CLOS assignment
    print("\n  MSR 0xC8F (PQR_ASSOC) — CLOS assignment per core:")
    for cbb_idx in range(NWP_NUM_CBBS):
        cbb = sv.socket0.getbypath(f"cbb{cbb_idx}")
        for compute in cbb.computes:
            for module in compute.modules:
                for core in module.cores:
                    path = core.target_info.path
                    try:
                        pqr = base.msr(0xC8F, path)
                        clos = (pqr >> 32) & 0x3
                        rmid = pqr & 0x3FF
                        print(f"    {path}: PQR=0x{pqr:016X} CLOS={clos} RMID={rmid}")
                    except Exception as e:
                        print(f"    {path}: ERROR — {e}")

    # Check MemClos TPMI configuration
    print("\n  TPMI MemClos registers (via OS mailbox):")
    for param in range(4):  # Check CLOS 0-3
        ret_data, ret_code = os_mailbox(command=OSMB_READ_PKG_CONFIG, sub_command=0x00, parameter=param)
        print(f"    CLOS{param}: data=0x{ret_data:08X} rc=0x{ret_code:02X}")


def test_memclos_throttling():
    """Test MemClos throttling status — HSD 22021970053."""
    print("\n" + "=" * 65)
    print("TEST: Mclos0 Throttling Status (HSD 22021970053)")
    print("=" * 65)

    # Read memory throttling status from TPMI
    print("\n  Memory PM TPMI registers:")
    try:
        # UFS status for memory fabric
        for imh in sv.socket0.imhs:
            ufs = imh.punit.ptpcfsms.ptpcfsms.ufs_status.read()
            ufs_f1 = imh.punit.ptpcfsms.ptpcfsms.ufs_status_fabric_1.read()
            ratio = ufs & 0x7F
            ratio_f1 = ufs_f1 & 0x7F
            print(f"    {imh.target_info.path} UFS ratio={ratio} fabric1_ratio={ratio_f1}")

            # Memory throttle count
            throttle_cnt = (ufs >> 32) & 0xFFFFFFFF
            throttle_cnt_f1 = (ufs_f1 >> 32) & 0xFFFFFFFF
            print(f"    {imh.target_info.path} throttle_cnt={throttle_cnt} f1_throttle_cnt={throttle_cnt_f1}")
    except Exception as e:
        print(f"    Memory UFS read: ERROR — {e}")

    # Read PLR for memory-related bits
    print("\n  PLR memory-related bits:")
    for cbb_idx in range(NWP_NUM_CBBS):
        try:
            sv.socket0.getbypath(f"cbb{cbb_idx}").base.tpmi.target_info["iosfsb_sai"] = 1
            mb_iface = sv.socket0.getbypath(f"cbb{cbb_idx}").base.tpmi.plr_mailbox_interface
            mb_iface.command.store(0)
            mb_iface.domain.store(0)
            mb_iface.id.store(0)
            mb_iface.run_busy.store(1)
            mb_iface.flush()
            wait(mb_iface)
            plr = sv.socket0.getbypath(f"cbb{cbb_idx}").base.tpmi.plr_mailbox_data.read()
            qos = (plr >> 8) & 1
            power = (plr >> 2) & 1
            fast_rapl = (plr >> 41) & 1
            print(f"    cbb{cbb_idx}: QOS={qos} POWER={power} FAST_RAPL={fast_rapl}")
        except Exception as e:
            print(f"    cbb{cbb_idx}: ERROR — {e}")


def test_mr4_cltt():
    """Test MR4-based CLTT — HSD 22021970051."""
    print("\n" + "=" * 65)
    print("TEST: MR4-based CLTT (HSD 22021970051)")
    print("=" * 65)

    # MR4 CLTT reads memory temperature via MR4 register
    # Check thermal status from TPMI OPC registers
    print("\n  OPC Package Thermal Status:")
    for cbb_idx in range(NWP_NUM_CBBS):
        try:
            therm = sv.socket0.getbypath(f"cbb{cbb_idx}").base.tpmi.opc_pkg_therm_status.read()
            thermal_status = therm & 1
            thermal_log = (therm >> 1) & 1
            power_lim_status = (therm >> 10) & 1
            print(f"    cbb{cbb_idx}: therm_status={thermal_status} therm_log={thermal_log} pwr_lim={power_lim_status}")
        except Exception as e:
            print(f"    cbb{cbb_idx}: ERROR — {e}")

    # Read memory temperature via OS mailbox if available
    print("\n  Memory temperature via OS mailbox:")
    # READ_PKG_CONFIG sub_command for memory thermal
    for subcmd in [0x00, 0x01, 0x10, 0x11]:
        ret_data, ret_code = os_mailbox(command=OSMB_READ_PKG_CONFIG, sub_command=subcmd, parameter=0)
        status = "OK" if ret_code == 0 else f"rc=0x{ret_code:02X}"
        print(f"    READ_PKG_CONFIG sub=0x{subcmd:02X}: data=0x{ret_data:08X} {status}")

    # Read DRAM RAPL for thermal throttle correlation
    print("\n  DRAM RAPL thermal correlation:")
    for cbb_idx in range(NWP_NUM_CBBS):
        cbb = sv.socket0.getbypath(f"cbb{cbb_idx}")
        for compute in cbb.computes:
            for module in compute.modules:
                core = list(module.cores)[0]  # first core
                path = core.target_info.path
                try:
                    dram_energy = base.msr(0x619, path)  # MSR_DRAM_ENERGY_STATUS
                    print(f"    {path}: DRAM_ENERGY=0x{dram_energy:016X}")
                    break  # one core per module is enough
                except Exception:
                    pass
                break
            break
