Inject core P-states via PEGA mailbox interface — U2P mailbox (CBB) for core/ring ratio, AUX mailbox (IMH) for mem/IO ratio. Sweep fixed ratios and random mode. Verify frequency transitions using UFS_STATUS.current_ratio (TPMI) and MSR 0x198 IA32_PERF_STATUS.

Record baseline values.

Step 2 — CBB PEGA P-state sweep:

For each target ratio in [8, 12, 16, 20, 24, 28, 32]:

a. Write PEGA CMD1 (CORE_RATIO[7:0]=ratio, RING_RATIO[23:16]=ratio) via U2P mailbox

b. Write PEGA CMD2 (CORE_VALID[0]=1, RING_VALID[2]=1) via U2P mailbox

c. Write PEGA CMD0 (Command[29:28]=0x2 P-state, Run_Busy[31]=1) via U2P mailbox

d. Wait for run_busy to clear

e. Read UFS_STATUS.current_ratio — verify it changed from previous

f. Read MSR 0x198 — verify core ratios reflect new P-state

Step 3 — IMH PEGA P-state sweep:

For each memgv in [8, 12, 15]:

a. Write PEGA CMD1 (MEM_VALID[1]=1) via AUX mailbox

b. Write PEGA CMD2 (MEM_RATIO[15:8]=memgv) via AUX mailbox

c. Write PEGA CMD0 (Command=0x2, Run_Busy=1) via AUX mailbox

d. Wait for completion

e. Read IMH UFS_STATUS.current_ratio — verify it changed

Step 4 — Random mode:

Set iagv=rand, meshgv=rand, rearm=rand, act2=rand.

Write PEGA CMD0/CMD1/CMD2 with random flags on all CBBs.

Wait, then read UFS_STATUS — verify current_ratio is non-zero.

def _imh_pstate(*, memgv = 0, iogv = 0, rearm = 0, act2 = 0):
    """ Injects PEGA pstate to the IMH die """

    # Module    Control
    #      0    IO Mesh frequency
    #      1    Memory Mesh frequency

    # IMH PEGA Cmd 1 Data Register for P States
    # ┌─────────┬────────┬─────────┬──────────────────────────────────────────────────────────┐
    # │  Bits   │ Access │ Default │                         Description                      │
    # ├─────────┼────────┼─────────┼──────────────────────────────────────────────────────────┤
    # │  7:0    │   RW   │    0    │ module_mask                                              │
    # │         │        │         │ Mask of modules to inject P-states. If bit n is set      │
    # │         │        │         │ then the target P-state value for module n will be       │
    # │         │        │         │ sent to that module. If bit n is cleared then the        │
    # │         │        │         │ target module is excluded from this injection.           │
    # ├─────────┼────────┼─────────┼──────────────────────────────────────────────────────────┤
    # │ 15:8    │   RW   │    0    │ random_state_target                                      │
    # │         │        │         │ Mask of modules for which random value will be used      │
    # │         │        │         │ instead of target P-state specified per module. Note     │
    # │         │        │         │ that bits set here will still be qualified based on      │
    # │         │        │         │ module_mask in order to determine whether a particular   │
    # │         │        │         │ module should be targeted or not.                        │
    # ├─────────┼────────┼─────────┼──────────────────────────────────────────────────────────┤
    # │ 16:16   │   RW   │    0    │ random_module_mask                                       │
    # │         │        │         │ Use random value instead of the value specified in       │
    # │         │        │         │ module_mask field.                                       │
    # ├─────────┼────────┼─────────┼──────────────────────────────────────────────────────────┤
    # │ 31:17   │   RW   │    0    │ Reserved                                                 │
    # └─────────┴────────┴─────────┴──────────────────────────────────────────────────────────┘

    data = 0

    if isinstance(iogv, int) and iogv >= 0:
        data = data | (0x1 << 0) # module_mask for io mesh

    if isinstance(memgv, int) and memgv >= 0:
        data = data | (0x1 << 1) # module_mask for memory mesh

    if str(iogv).lower() == "rand":
        data = data | (0x1 << 0) # module_mask for io mesh
        data = data | (0x1 << 8) # random_state_target for io mesh

    if str(memgv).lower() == "rand":
        data = data | (0x1 << 1) # module_mask for memory mesh
        data = data | (0x1 << 9) # random_state_target for memory mesh


    aux_mailbox(command=_pega_dict["WRITE_PEGA_CMD1"], data=data)

    # IMH PEGA Cmd 2 Data Register for P States
    # Note: On DMR, Primecode treats the values for Module 2 and 3 as reserved and ignore those fields
    #
    # ┌─────────┬────────┬─────────┬──────────────────────────────────────────────────────────┐
    # │  Bits   │ Access │ Default │                         Description                      │
    # ├─────────┼────────┼─────────┼──────────────────────────────────────────────────────────┤
    # │  7:0    │   RW   │    0    │ MODULE0_RATIO                                            │
    # │         │        │         │ Target ratio to be injected for module 0                 │
    # ├─────────┼────────┼─────────┼──────────────────────────────────────────────────────────┤
    # │ 15:8    │   RW   │    0    │ MODULE1_RATIO                                            │
    # │         │        │         │ Target ratio to be injected for module 1                 │
    # ├─────────┼────────┼─────────┼──────────────────────────────────────────────────────────┤
    # │ 23:16   │   RW   │    0    │ MODULE2_RATIO                                            │
    # │         │        │         │ Target ratio to be injected for module 2                 │

    data = 0

    if isinstance(iogv, int) and 0 <= iogv <= 0xFF: # MODULE0_RATIO
        data = data | (iogv << 0)

    if isinstance(memgv, int) and 0 <= memgv <= 0xFF: # MODULE1_RATIO
        data = data | (memgv << 8)

    aux_mailbox(command=_pega_dict["WRITE_PEGA_CMD2"], data=data)

    # IMH PEGA Cmd 3 Data Register for P States
    # Note: On DMR, Primecode treats the values for Module 4 through 7 as reserved and ignore those fields
    #
    # ┌─────────┬────────┬─────────┬──────────────────────────────────────────────────────────┐
    # │  Bits   │ Access │ Default │                         Description                      │
    # ├─────────┼────────┼─────────┼──────────────────────────────────────────────────────────┤

    # ├─────────┼────────┼─────────┼──────────────────────────────────────────────────────────┤

    # │         │        │         │ Target ratio to be injected for module 6                 │
    # ├─────────┼────────┼─────────┼──────────────────────────────────────────────────────────┤
    # │ 31:24   │   RW   │    0    │ MODULE7_RATIO                                            │
    # │         │        │         │ Target ratio to be injected for module 7                 │
    # └─────────┴────────┴─────────┴──────────────────────────────────────────────────────────┘

    # We can skip this step, as primecode will ignore anything we set here
    # data = 0

    # aux_mailbox(command=_pega_dict["WRITE_PEGA_CMD3"], data=data)

    # PEGA Cmd 0 Data Register
    #
    # ┌─────────┬────────┬─────────┬──────────────────────────────────────────────────────────┐

    # ├─────────┼────────┼─────────┼──────────────────────────────────────────────────────────┤

    # │         │        │         │ Latency to wait after one command before executing it    │
    # │         │        │         │ again. When command has 2nd action enabled this is the   │
    # │         │        │         │ time from when the 2nd action is issued until the 1st    │
    # │         │        │         │ action is issued again. Units specified in               │
    # │         │        │         │ rearm_latency_units field. Value of zero indicates that  │
    # │         │        │         │ rearming action is disabled.                             │
    # ├─────────┼────────┼─────────┼──────────────────────────────────────────────────────────┤
    # │ 15:8    │  RWV   │    0    │ Action2 Latency                                          │
    # │         │        │         │ Latency to wait after initial event before injecting     │
    # │         │        │         │ the second action associated with this event (unforce    │
    # │         │        │         │ P-state for P-state). Units specified in                 │
    # │         │        │         │ wake_latency_units field. Value of zero indicates that   │
    # │         │        │         │ 2nd action is disabled.                                  │
    # ├─────────┼────────┼─────────┼──────────────────────────────────────────────────────────┤
    # │ 17:16   │  RWV   │    0    │ Rearm Latency Units                                      │
    # │         │        │         │ Units for rearm timeout.                                 │
    # │         │        │         │ Value Definition:                                        │
    # │         │        │         │   0: units of 640 ns (0.64 us)                           │
    # │         │        │         │   1: units of 10240 ns (~10.2 us)                        │
    # │         │        │         │   2: units of 163840 ns (~164 us)                        │
    # │         │        │         │   3: units of 2621440 ns (~2.6 ms)                       │
    # ├─────────┼────────┼─────────┼──────────────────────────────────────────────────────────┤

    # │         │        │         │ Unit for 2nd action timeout.                             │
    # │         │        │         │ Value Definition:                                        │
    # │         │        │         │   0: units of 640 ns (0.64 us)                           │
    # │         │        │         │   1: units of 10240 ns (~10.2 us)                        │
    # │         │        │         │   2: units of 163840 ns (~164 us)                        │
    # │         │        │         │   3: units of 2621440 ns (~2.6 ms)                       │
    # ├─────────┼────────┼─────────┼──────────────────────────────────────────────────────────┤
    # │ 20:20   │  RWV   │    0    │ Random Rearm Latency                                     │
    # │         │        │         │ Use random 8 bit value instead of the value specified    │
    # │         │        │         │ in rearm_latency field. Units are still specified in     │
    # │         │        │         │ rearm_latency_units field. Note that rearm latency       │

    # │         │        │         │ If set, use random 8 bit value instead of the value      │
    # │         │        │         │ specified in action2_latency field. Units are still      │
    # │         │        │         │ specified in action2_latency_units field. Note that      │
    # │         │        │         │ action2 latency should still be non-zero to indicate     │

    # │ 27:22   │   RO   │    0    │ Reserved                                                 │
    # ├─────────┼────────┼─────────┼──────────────────────────────────────────────────────────┤
    # │ 30:28   │  RWV   │    0    │ Command                                                  │
    # │         │        │         │ A. Type of PEGA command to trigger (User writes)         │
    # │         │        │         │ Value Definition:                                        │
    # │         │        │         │   0: No command                                          │
    # │         │        │         │   1: invalid (was C-state injection)                     │
    # │         │        │         │   2: P-state injection                                   │
    # │         │        │         │   others: Invalid                                        │
    # │         │        │         │ B. Error Code (Updated by Primecode as return status)    │
    # │         │        │         │ Value Definition:                                        │
    # │         │        │         │   0: No Error                                            │
    # │         │        │         │   1: Invalid Command                                     │
    # │         │        │         │ On an invalid command, this field will updated to 0x1    │
    # │         │        │         │ and all other fields will be cleared. Additionally the   │
    # │         │        │         │ Mailbox Interface return status will indicate            │
    # │         │        │         │ INVALID_PARAMETERS                                       │
    # ├─────────┼────────┼─────────┼──────────────────────────────────────────────────────────┤
    # │ 31:31   │  RWV   │    0    │ Run_Busy                                                 │
    # │         │        │         │ Set by user to activate PEGA after setting up the other  │
    # │         │        │         │ registers - this will trigger PCode to perform the       │
    # │         │        │         │ requested PEGA action. This bit will be cleared by       │
    # │         │        │         │ PCode after sampling is complete and completion code     │
    # │         │        │         │ has been set. User should not change PEGA registers      │
    # │         │        │         │ until PCode has cleared run_busy or they risk breaking   │
    # │         │        │         │ atomicity of the command and/or having a command         │
    # │         │        │         │ dropped. This will not be protected against in HW.       │
    # │         │        │         │ Writing CMD0 with run_busy == 0 will result in an        │
    # │         │        │         │ INVALID_PARAMETER error response. In this case, the      │
    # │         │        │         │ Data field will be preserved for help in debugging       │
    # │         │        │         │ the cause.                                               │
    # └─────────┴────────┴─────────┴──────────────────────────────────────────────────────────┘

    data = 0

    if isinstance(rearm, (int, float)) and rearm >= 0:
        rearm_lat, rearm_units = calc_units(rearm)
        data = data | (rearm_lat << 0)    # Rearm Latency
        data = data | (rearm_units << 16) # Rearm units
    elif str(rearm).lower() == "rand":
        data = data | (1 << 0)  # Latency must be non-zero even if random
        data = data | (1 << 20) # Random Rearm Latency
    if isinstance(act2, (int, float)) and act2 >= 0:
        act2_lat, act2_units = calc_units(act2)
        data = data | (act2_lat << 8)    # Action2 Latency
        data = data | (act2_units << 18) # Action2 Latency Units
    elif str(act2).lower() == "rand":
        data = data | (1 << 8)  # Latency must be non-zero even if random
        data = data | (1 << 21) # Random Action2 Latency

    data = data | (0x2 << 28) # Set P-state injection
    data = data | (0x1 << 31) # Set Run Busy

    aux_mailbox(command=_pega_dict["WRITE_PEGA_CMD0"], data=data)

    return 0

        data = data | (1 << 10) # RANDOM_DOMAIN_VALID

    # │         │        │         │   1: units of 10240 ns (~10.2 us)                        │
    # │         │        │         │   2: units of 163840 ns (~164 us)                        │

    # │         │        │         │ A. Type of PEGA command to trigger (User writes)         │
    # │         │        │         │ Value Definition:                                        │
    # │         │        │         │   0: No command                                          │
    # │         │        │         │   1: C-state injection                                   │
    # │         │        │         │   2: P-state injection                                   │
    # │         │        │         │   others: Invalid                                        │
    # │         │        │         │ B. Error Code (Pcode writes)                             │
    # │         │        │         │ Value Definition:                                        │
    # │         │        │         │   0: No Error                                            │
    # │         │        │         │   1: Invalid Command, this field will updated to 0x1     │
    # │         │        │         │       and all other fields will be cleared.              │
    # │         │        │         │       Additionally the Mailbox Interface return status   │
    # │         │        │         │       will indicate INVALID_PARAMETERS.                  │
    # ├─────────┼────────┼─────────┼──────────────────────────────────────────────────────────┤
    # │ 31:31   │  RWV   │    0    │ Run_Busy                                                 │
    # │         │        │         │ Set by user to activate PEGA after setting up the other  │
    # │         │        │         │ registers - this will trigger PCode to perform the       │
    # │         │        │         │ requested PEGA action. This bit will be cleared by       │
    # │         │        │         │ PCode after sampling is complete and completion code     │
    # │         │        │         │ has been set. User should not change PEGA registers      │
    # │         │        │         │ until PCode has cleared run_busy or they risk breaking   │
    # │         │        │         │ atomicity of the command and/or having a command         │
    # │         │        │         │ dropped. This will not be protected against in HW.       │
    # │         │        │         │ Writing CMD0 with run_busy == 0 will result in an        │
    # │         │        │         │ INVALID_PARAMETER error response. In this case, the      │
    # │         │        │         │ Data field will be preserved for help in debugging       │
    # │         │        │         │ the cause.                                               │
    # └─────────┴────────┴─────────┴──────────────────────────────────────────────────────────┘

    data = 0

    if isinstance(rearm, (int, float)) and rearm >= 0:
        rearm_lat, rearm_units = calc_units(rearm)
        data = data | (rearm_lat << 0)    # Rearm Latency
        data = data | (rearm_units << 16) # Rearm units
    elif str(rearm).lower() == "rand":
        data = data | (1 << 0)  # Latency must be non-zero even if random
        data = data | (1 << 20) # Random Rearm Latency
    if isinstance(act2, (int, float)) and act2 >= 0:
        act2_lat, act2_units = calc_units(act2)
        data = data | (act2_lat << 8)    # Action2 Latency
        data = data | (act2_units << 18) # Action2 Latency Units
    elif str(act2).lower() == "rand":
        data = data | (1 << 8)  # Latency must be non-zero even if random
        data = data | (1 << 21) # Random Action2 Latency

    data = data | (0x2 << 28) # Set P-state injection
    data = data | (0x1 << 31) # Set Run Busy

    u2p_mailbox(command=_pega_dict["WRITE_PEGA_CMD0"], data=data)

    return 0


def calc_units(latency_ms = 0):
    """Calculate latency value and unit encoding for PEGA timing registers.
    
    Args:
        latencyMs (float): Latency in milliseconds
        
    Returns:
        tuple: (latency_value, unit_encoding) where:
            - latency_value (int): 8-bit value for latency field (1-255)
            - unit_encoding (int): 2-bit unit encoding (0-3)
        
    Unit Encodings:
        0: 640 ns (0.64 us) units
        1: 10240 ns (~10.2 us) units  
        2: 163840 ns (~164 us) units
        3: 2621440 ns (~2.6 ms) units
    """
    if latency_ms <= 0:
        return 0, 0

    # PEGA unit definitions in nanoseconds
    unit_scales_ns = [640, 10240, 163840, 2621440]
    max_latency_value = 255  # Maximum we can find in the 8 bits
    min_latency_value = 1    # Bare minimum latency, 640ns

    latency_ns = latency_ms * 1000000  # Convert ms to ns

    # Find the most precise unit that can represent this latency
    for unit_encoding, unit_scale_ns in enumerate(unit_scales_ns):
        latency_value = latency_ns / unit_scale_ns

        if min_latency_value <= latency_value <= max_latency_value:
            return int(latency_value), unit_encoding

    # If latency is too large for all units, clamp to maximum representable value
    # Using the largest unit (index 3) at maximum value
    max_representable_ns = unit_scales_ns[3] * max_latency_value
    if latency_ns > max_representable_ns:
        return max_latency_value, 3

    # If latency is too small for the finest unit, use minimum value with finest unit
    return min_latency_value, 0


def aux_mailbox(*, command=0, param=0, data=0):
    """ Helper function to do the aux mailbox injection """

    for socket in namednodes.sv.sockets:
        for imh in socket.imhs:
            aux_mailbox_interface = imh.punit.ptpcfsms.ptpcfsms.aux_mailbox_interface
            aux_mailbox_data = imh.punit.ptpcfsms.ptpcfsms.aux_mailbox_data

            wait(aux_mailbox_interface)

            print(f"PM_INFO :: Writing {socket.name}.{imh.name} Aux Mailbox Interface " \
                  f"to 0x{(command + (param<<8) + (1 << 63)):x}")
            print(f"PM_INFO :: Writing {socket.name}.{imh.name} Aux Mailbox Data to 0x{data:x}")

            aux_mailbox_data.write(data)
            aux_mailbox_interface.command.store(command)
            aux_mailbox_interface.param.store(param)
            aux_mailbox_interface.run_busy.store(1)
            aux_mailbox_interface.flush()

            wait(aux_mailbox_interface)

            print(f"PM_INFO :: Reading {socket.name}.{imh.name} Aux Mailbox return code: " \
                  f"{aux_mailbox_interface.read()}")
    return 0


def u2p_mailbox(*, socket=0, cbb=0, command=0, param1=0, param2=0, data=0):
    """ Helper function to do the U2P mailbox injection """

    socket_node = namednodes.sv.socket.getBySocket(socket)

    u2p_mailbox_interface = \
        socket_node.getbypath(f"cbb{cbb}").base.punit_regs.punit_gpsb.gpsb_infvnn_crs.ucode_pcode_mailbox_interface["fid_0"]
    u2p_mailbox_data = \
        socket_node.getbypath(f"cbb{cbb}").base.punit_regs.punit_gpsb.gpsb_infvnn_crs.ucode_pcode_mailbox_data["fid_0"]

    wait(u2p_mailbox_interface)

    print(f"PM_INFO :: Writing socket{socket}.cbb{cbb} U2P Mailbox Interface to 0x{(command + (param1<<8) + (param2<<16) + (1 << 31)):x}")
    print(f"PM_INFO :: Writing socket{socket}.cbb{cbb} U2P Mailbox Data to 0x{data:x}")

    u2p_mailbox_data.write(data)
    u2p_mailbox_interface.command.store(command)
    u2p_mailbox_interface.param1.store(param1)
    u2p_mailbox_interface.param2.store(param2)
    u2p_mailbox_interface.run_busy.store(1)
    u2p_mailbox_interface.flush()

    wait(u2p_mailbox_interface)

    print(f"PM_INFO :: Reading socket.{socket}.cbb{cbb} U2P Mailbox return code: {u2p_mailbox_interface.read()}")

    return 0


def wait(interface = None):
    """ Inject wait-for-cycle commands into emulation """
    wait_time = 0
    wait_cycles = 100000
    wait_timeout = 20

    while interface.run_busy.read():
        if ACCESS_METHOD == "stub":
            interface.write(0)
        elif ACCESS_METHOD == "emulation":
            cli.run_command(f"emu.engine.wait-for-cycle -relative {wait_cycles}")
        elif ACCESS_METHOD == "simics":
            cli.run_command(f"run-cycles {wait_cycles}")
        else:
            time.sleep(1)

        wait_time += 1
        wait_cycles += 10000

        if wait_time > wait_timeout:
            return -1
    return 0
